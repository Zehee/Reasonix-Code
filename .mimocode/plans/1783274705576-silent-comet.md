## Step 3: Create desktop/vite.config.ts

Point Vite to build from `dashboard/src/` instead of `desktop/src/`:

```ts
import { defineConfig } from "vite";
import { resolve } from "path";

export default defineConfig({
  root: resolve(__dirname, ".."),  // project root
  publicDir: resolve(__dirname, "..", "dashboard", "public"),
  build: {
    outDir: resolve(__dirname, "dist"),
    emptyOutDir: true,
    rollupOptions: {
      input: resolve(__dirname, "..", "dashboard", "index.html"),
    },
  },
  resolve: {
    alias: {
      "@reasonix/core-utils": resolve(__dirname, "..", "packages", "core-utils", "src"),
    },
  },
  server: {
    port: 1420,
    strictPort: true,
  },
});
```
